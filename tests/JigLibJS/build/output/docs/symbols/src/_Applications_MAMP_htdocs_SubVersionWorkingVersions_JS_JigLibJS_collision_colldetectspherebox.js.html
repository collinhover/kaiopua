 21 (function(jigLib){
 22 	var Vector3DUtil=jigLib.Vector3DUtil;
 23 	var JNumber3D=jigLib.JNumber3D;
 24 	var JConfig=jigLib.JConfig;
 25 	var MaterialProperties=jigLib.MaterialProperties;
 26 	var CollPointInfo=jigLib.CollPointInfo;
 27 	var CollisionInfo=jigLib.CollisionInfo;
 28 
 29 	/**
 30 	 * @author Muzer(muzerly@gmail.com)
 31 	 * 
 32 	 * @name CollDetectSphereBox
 33 	 * @class CollDetectSphereBox handles collisions between spheres and boxes
 34 	 * @extends CollDetectFunctor
 35 	 * @requires CollDetectInfo
 36 	 * @requires CollPointInfo
 37 	 * @requires CollisionInfo
 38 	 * @requires Vector3DUtil
 39 	 * @requires JNumber3D
 40 	 * @requires JConfig
 41 	 * @requires MaterialProperties
 42 	 * @constructor
 43 	 **/
 44 	var CollDetectSphereBox=function(){
 45 		this.name = "SphereBox";
 46 		this.type0 = "SPHERE";
 47 		this.type1 = "BOX";
 48 	};
 49 	jigLib.extend(CollDetectSphereBox,jigLib.CollDetectFunctor);
 50 	
 51 				
 52 	/**
 53 	 * @function collDetect detects a collision and updates the info parameter
 54 	 * @param {CollDetectInfo} info
 55 	 * @param {array} collArray
 56 	 * @type void
 57 	 **/
 58 	CollDetectSphereBox.prototype.collDetect=function(info, collArr){
 59 		var tempBody;
 60 		if(info.body0.get_type()=="BOX") {
 61 			tempBody=info.body0;
 62 			info.body0=info.body1;
 63 			info.body1=tempBody;
 64 		}
 65 						
 66 		var sphere = info.body0;
 67 		var box = info.body1;

		
 68 		if (!sphere.hitTestObject3D(box)) 
 69 			return;
 70 
 71 		if (JConfig.aabbDetection && !sphere.get_boundingBox().overlapTest(box.get_boundingBox())) 
 72 			return;
 73 		
 74 		//var spherePos:Vector3D = sphere.get_oldState().position;
 75 		//var boxPos:Vector3D = box.get_oldState().position;
 76 
 77 		var oldBoxPoint={};
 78 		var newBoxPoint={};
 79 						
 80 		var oldDist = box.getDistanceToPoint(box.get_oldState(), oldBoxPoint, sphere.get_oldState().position);
 81 		var newDist = box.getDistanceToPoint(box.get_currentState(), newBoxPoint, sphere.get_currentState().position);
 82 						
 83 		var oldDepth = sphere.get_radius() - oldDist;
 84 		var newDepth = sphere.get_radius() - newDist;
 85 		if (Math.max(oldDepth, newDepth) > -JConfig.collToll) {
 86 			var dir;
 87 			var collPts = [];
 88 			if (oldDist < -JNumber3D.NUM_TINY) {
 89 				dir = Vector3DUtil.subtract(Vector3DUtil.subtract(oldBoxPoint.pos, 
 90 																  sphere.get_oldState().position), 
 91 											oldBoxPoint.pos);
 92 				Vector3DUtil.normalize(dir);
 93 			}else if (oldDist > JNumber3D.NUM_TINY) {
 94 				dir = Vector3DUtil.subtract(sphere.get_oldState().position, oldBoxPoint.pos);
 95 				Vector3DUtil.normalize(dir);
 96 			}else{
 97 				dir = Vector3DUtil.subtract(sphere.get_oldState().position, box.get_oldState().position);
 98 				Vector3DUtil.normalize(dir);
 99 			}
100 								
101 			var cpInfo = new CollPointInfo();
102 			cpInfo.r0 = Vector3DUtil.subtract(oldBoxPoint.pos, sphere.get_oldState().position);
103 			cpInfo.r1 = Vector3DUtil.subtract(oldBoxPoint.pos, box.get_oldState().position);
104 			cpInfo.initialPenetration = oldDepth;
105 			collPts.push(cpInfo);
106 								
107 			var collInfo=new CollisionInfo();
108 			collInfo.objInfo=info;
109 			collInfo.dirToBody = dir;
110 			collInfo.pointInfo = collPts;
111 								
112 			var mat = new MaterialProperties();
113 			mat.set_restitution(Math.sqrt(sphere.get_material().get_restitution() * box.get_material().get_restitution()));
114 			mat.set_friction(Math.sqrt(sphere.get_material().get_friction() * box.get_material().get_friction()));
115 			collInfo.mat = mat;
116 			collArr.push(collInfo);
117 								
118 			info.body0.collisions.push(collInfo);
119 			info.body1.collisions.push(collInfo);
120 		}
121 	};
122 	
123 	jigLib.CollDetectSphereBox=CollDetectSphereBox;
124 	
125 })(jigLib);
