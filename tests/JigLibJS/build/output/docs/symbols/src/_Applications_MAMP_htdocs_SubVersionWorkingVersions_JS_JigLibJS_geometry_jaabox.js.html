 27 	/**
 28 	 * @author Muzer(muzerly@gmail.com)
 29 	 * 
 30 	 * @name JAABox
 31 	 * @class JAABox an axis aligned box
 32 	 * @requires Vector3DUtil
 33 	 * @requires JNumber3D
 34 	 * @property {array} minPos a 3D vector
 35 	 * @property {array} maxPos a 3D vector
 36 	 * @constructor
 37 	 * @param {array} minPos a 3D vector
 38 	 * @param {array} maxPos a 3D vector
 39 	 **/
 40 	var JAABox=function(minPos, maxPos) {
 41 		if(minPos){
 42 			this._minPos = minPos.slice(0);
 43 			this._maxPos = maxPos.slice(0);
 44 		}
 45 	};
 46

 51 	 * @function get_minPos getter for _minPos

 56 	};

 84 	/**
 85 	 * @function get_sideLengths determines the side lengths of the JAABox
 86 	 * @returns the side lengths expressed as 3D vector
 87 	 * @type array
 88 	 **/
 89 	JAABox.prototype.get_sideLengths=function() {
 90 		var pos = this._maxPos.slice(0);
 91 		Vector3DUtil.subtract(pos, this._minPos);
 92 		return pos;
 93 	};

 95 	/**
 96 	 * @function get_centrePos determines the center point of the JAABox
 97 	 * @returns the center point expressed as 3D vector
 98 	 * @type array
 99 	 **/
100 	JAABox.prototype.get_centrePos=function(){
101 		var pos = this._minPos.slice(0);
102 		return JNumber3D.getScaleVector(Vector3DUtil.add(pos, this._maxPos), 0.5);
103 	};

106 	JAABox.prototype.getAllPoints=function(){
107 		var center,halfSide;
108 		var points;
109 		center = this.get_centrePos();
110 		halfSide = this.get_sideLengths().slice(0);
111 		Vector3DUtil.scaleBy(halfSide, 0.5);
112 		points = [];
113 		points[0] = Vector3DUtil.add(center,[halfSide[0], -halfSide[1], halfSide[2]]);
114 		points[1] = Vector3DUtil.add(center,[halfSide[0], halfSide[1], halfSide[2]]);
115 		points[2] = Vector3DUtil.add(center,[-halfSide[0], -halfSide[1], halfSide[2]]);
116 		points[3] = Vector3DUtil.add(center,[-halfSide[0], halfSide[1], halfSide[2]]);
117 		points[4] = Vector3DUtil.add(center,[-halfSide[0], -halfSide[1], -halfSide[2]]);
118 		points[5] = Vector3DUtil.add(center,[-halfSide[0], halfSide[1], -halfSide[2]]);
119 		points[6] = Vector3DUtil.add(center,[halfSide[0], -halfSide[1], -halfSide[2]]);
120 		points[7] = Vector3DUtil.add(center,[halfSide[0], halfSide[1], -halfSide[2]]);
121                         
122 		return points;
123 	}

162 	/**
163 	 * @function addPoint  
164 	 * @param {array} pos a 3D vector
165 	 * @type void
166 	 **/
167 	JAABox.prototype.addPoint=function(pos){
168 		var _minPos=this._minPos;
169 		var _maxPos=this._maxPos;
170 		if (pos[0] < _minPos[0]) _minPos[0] = pos[0] - JNumber3D.NUM_TINY;
171 		if (pos[0] > _maxPos[0]) _maxPos[0] = pos[0] + JNumber3D.NUM_TINY;
172 		if (pos[1] < _minPos[1]) _minPos[1] = pos[1] - JNumber3D.NUM_TINY;
173 		if (pos[1] > _maxPos[1]) _maxPos[1] = pos[1] + JNumber3D.NUM_TINY;
174 		if (pos[2] < _minPos[2]) _minPos[2] = pos[2] - JNumber3D.NUM_TINY;
175 		if (pos[2] > _maxPos[2]) _maxPos[2] = pos[2] + JNumber3D.NUM_TINY;
176 	};

196 	 * @function addSphere
197 	 * @param {JSphere} sphere 
198 	 * @type void
199 	 **/
200 	JAABox.prototype.addSphere=function(sphere){
201 		var _minPos=this._minPos;
202 		var _maxPos=this._maxPos;
203 		if (sphere.get_currentState().position[0] - sphere.get_radius() < _minPos[0]) 
204 			_minPos[0] = (sphere.get_currentState().position[0] - sphere.get_radius()) - JNumber3D.NUM_TINY;
205 
206 		if (sphere.get_currentState().position[0] + sphere.get_radius() > _maxPos[0]) 
207 			_maxPos[0] = (sphere.get_currentState().position[0] + sphere.get_radius()) + JNumber3D.NUM_TINY;
208 
209 		if (sphere.get_currentState().position[1] - sphere.get_radius() < _minPos[1]) 
210 			_minPos[1] = (sphere.get_currentState().position[1] - sphere.get_radius()) - JNumber3D.NUM_TINY;
211 
212 		if (sphere.get_currentState().position[1] + sphere.get_radius() > _maxPos[1]) 
213 			_maxPos[1] = (sphere.get_currentState().position[1] + sphere.get_radius()) + JNumber3D.NUM_TINY;
214 
215 		if (sphere.get_currentState().position[2] - sphere.get_radius() < _minPos[2]) 
216 			_minPos[2] = (sphere.get_currentState().position[2] - sphere.get_radius()) - JNumber3D.NUM_TINY;
217 
218 		if (sphere.get_currentState().position[2] + sphere.get_radius() > _maxPos[2]) 
219 			_maxPos[2] = (sphere.get_currentState().position[2] + sphere.get_radius()) + JNumber3D.NUM_TINY;
220 	};

223 	 * @function addCapsule  
224 	 * @param {JCapsule} capsule 
225 	 * @type void
226 	 **/
227 	JAABox.prototype.addCapsule=function(capsule){
228 		var pos= capsule.getBottomPos(capsule.get_currentState());
229 		var _minPos=this._minPos;
230 		var _maxPos=this._maxPos;
231 		if (pos[0] - capsule.get_radius() < _minPos[0]) 
232 			_minPos[0] = (pos[0] - capsule.get_radius()) - JNumber3D.NUM_TINY;
233 
234 		if (pos[0] + capsule.get_radius() > _maxPos[0]) 
235 			_maxPos[0] = (pos[0] + capsule.get_radius()) + JNumber3D.NUM_TINY;
236 
237 		if (pos[1] - capsule.get_radius() < _minPos[1]) 
238 			_minPos[1] = (pos[1] - capsule.get_radius()) - JNumber3D.NUM_TINY;
239 
240 		if (pos[1] + capsule.get_radius() > _maxPos[1]) 
241 			_maxPos[1] = (pos[1] + capsule.get_radius()) + JNumber3D.NUM_TINY;
242 
243 		if (pos[2] - capsule.get_radius() < _minPos[2]) 
244 			_minPos[2] = (pos[2] - capsule.get_radius()) - JNumber3D.NUM_TINY;
245 
246 		if (pos[2] + capsule.get_radius() > _maxPos[2]) 
247 			_maxPos[2] = (pos[2] + capsule.get_radius()) + JNumber3D.NUM_TINY;
248 
249 		pos = capsule.getEndPos(capsule.get_currentState());
250 
251 		if (pos[0] - capsule.get_radius() < _minPos[0]) 
252 			_minPos[0] = (pos[0] - capsule.get_radius()) - JNumber3D.NUM_TINY;
253 
254 		if (pos[0] + capsule.get_radius() > _maxPos[0]) 
255 			_maxPos[0] = (pos[0] + capsule.get_radius()) + JNumber3D.NUM_TINY;
256 
257 		if (pos[1] - capsule.get_radius() < _minPos[1]) 
258 			_minPos[1] = (pos[1] - capsule.get_radius()) - JNumber3D.NUM_TINY;
259 
260 		if (pos[1] + capsule.get_radius() > _maxPos[1]) 
261 			_maxPos[1] = (pos[1] + capsule.get_radius()) + JNumber3D.NUM_TINY;
262 
263 		if (pos[2] - capsule.get_radius() < _minPos[2]) 
264 			_minPos[2] = (pos[2] - capsule.get_radius()) - JNumber3D.NUM_TINY;
265 
266 		if (pos[2] + capsule.get_radius() > _maxPos[2]) 
267 			_maxPos[2] = (pos[2] + capsule.get_radius()) + JNumber3D.NUM_TINY;
268 	};

280 	/**
281 	 * @function overlapTest tests for an overlap between 2 boxes  
282 	 * @param {JAABox} box 
283 	 * @type boolean
284 	 **/
285 	JAABox.prototype.overlapTest=function(box){
286 		var _minPos=this._minPos;
287 		var _maxPos=this._maxPos;
288 		return ((_minPos[2] >= box.get_maxPos()[2]) ||
289 				(_maxPos[2] <= box.get_minPos()[2]) ||
290 				(_minPos[1] >= box.get_maxPos()[1]) ||
291 				(_maxPos[1] <= box.get_minPos()[1]) ||
292 				(_minPos[0] >= box.get_maxPos()[0]) ||
293 				(_maxPos[0] <= box.get_minPos()[0])) ? false : true;
294 	};

297 	 * @function isPointInside tests if a given point lies inside this JAABox  
298 	 * @param {array} pos a 3D vector
299 	 * @type boolean
300 	 **/
301 	JAABox.prototype.isPointInside=function(pos){
302 		var _minPos=this._minPos;
303 		var _maxPos=this._maxPos;
304 		return ((pos[0] >= _minPos[0]) && 
305 				(pos[0] <= _maxPos[0]) && 
306 				(pos[1] >= _minPos[1]) && 
307 				(pos[1] <= _maxPos[1]) && 
308 				(pos[2] >= _minPos[2]) && 
309 				(pos[2] <= _maxPos[2]));
310 	};
