 21 (function(jigLib){
 22 	var Vector3DUtil=jigLib.Vector3DUtil;
 23 	var JNumber3D=jigLib.JNumber3D;
 24 	var JConstraint=jigLib.JConstraint;
 25 	var JConfig=jigLib.JConfig;
 26 	var JCapsule=jigLib.JCapsule;
 27 	var JTerrain=jigLib.JTerrain;
 28 	var MaterialProperties=jigLib.MaterialProperties;
 29 	var RigidBody=jigLib.RigidBody;
 30 	
 31 	/**
 32 	 * @author Muzer(muzerly@gmail.com)
 33 	 * 
 34 	 * @name CollDetectCapsuleTerrain
 35 	 * @class CollDetectCapsuleTerrain handles collisions between capsules and terrain
 36 	 * @extends CollDetectFunctor
 37 	 * @requires CollDetectInfo
 38 	 * @requires CollPointInfo
 39 	 * @requires Vector3DUtil
 40 	 * @requires JNumber3D
 41 	 * @requires CollisionInfo
 42 	 * @requires MaterialProperties
 43 	 * @constructor
 44 	 **/
 45 	var CollDetectCapsuleTerrain=function(){
 46 		this.name = "BoxTerrain";
 47 		this.type0 = "CAPSULE";
 48 		this.type1 = "TERRAIN";
 49 	};
 50 	jigLib.extend(CollDetectCapsuleTerrain,jigLib.CollDetectFunctor);
 51 
 52 	/**
 53 	 * @function collDetect detects a collision and updates the info parameter
 54 	 * @param {CollDetectInfo} info
 55 	 * @param {array} collArray
 56 	 * @type void
 57 	 **/
 58 	CollDetectCapsuleTerrain.prototype.collDetect=function(info, collArr){
 59 		var tempBody;
 60 		if (info.body0.type == "TERRAIN"){
 61 			tempBody = info.body0;
 62 			info.body0 = info.body1;
 63 			info.body1 = tempBody;
 64 		}
 65 						
 66 		var capsule = info.body0;
 67 		var terrain = info.body1;
 68 						
 69 		var collPts = [];
 70 		var cpInfo;
 71 						
 72 		var averageNormal= [0,0,0,0];
 73 		var pos1 = capsule.getBottomPos(capsule.oldState);
 74 		var pos2 = capsule.getBottomPos(capsule.currentState);
 75 		var obj1= terrain.getHeightAndNormalByPoint(pos1);
 76 		var obj2 = terrain.getHeightAndNormalByPoint(pos2);
 77 		if (Math.min(obj1.height, obj2.height) < JConfig.collToll + capsule.radius) {
 78 			var oldDepth = capsule.radius - obj1.height;
 79 			var worldPos = Vector3DUtil.subtract(pos1, JNumber3D.getScaleVector(obj2.normal, capsule.radius));
 80 			cpInfo = new CollPointInfo();
 81 			cpInfo.r0 = Vector3DUtil.subtract(worldPos, capsule.oldState.position);
 82 			cpInfo.r1 = Vector3DUtil.subtract(worldPos, terrain.oldState.position);
 83 			cpInfo.initialPenetration = oldDepth;
 84 			collPts.push(cpInfo);
 85 			averageNormal = Vector3DUtil.add(averageNormal, obj2.normal);
 86 		}
 87 						
 88 		pos1 = capsule.getEndPos(capsule.oldState);
 89 		pos2 = capsule.getEndPos(capsule.currentState);
 90 		obj1 = terrain.getHeightAndNormalByPoint(pos1);
 91 		obj2 = terrain.getHeightAndNormalByPoint(pos2);
 92 		if (Math.min(obj1.height, obj2.height) < JConfig.collToll + capsule.radius) {
 93 			oldDepth = capsule.radius - obj1.height;
 94 			worldPos = Vector3DUtil.subtract(pos1, JNumber3D.getScaleVector(obj2.normal, capsule.radius));
 95 			cpInfo = new CollPointInfo();
 96 			cpInfo.r0 = Vector3DUtil.subtract(worldPos, capsule.oldState.position);
 97 			cpInfo.r1 = Vector3DUtil.subtract(worldPos, terrain.oldState.position);
 98 			cpInfo.initialPenetration = oldDepth;
 99 			collPts.push(cpInfo);
100 			averageNormal = Vector3DUtil.add(averageNormal, obj2.normal);
101 		}
102 						
103 		if (collPts.length > 0){
104 			Vector3DUtil.normalize(averageNormal);
105 			var collInfo = new CollisionInfo();
106 			collInfo.objInfo = info;
107 			collInfo.dirToBody = averageNormal;
108 			collInfo.pointInfo = collPts;
109 
110 			var mat = new MaterialProperties();
111 			mat.restitution = Math.sqrt(capsule.material.restitution * terrain.material.restitution);
112 			mat.friction = Math.sqrt(capsule.material.friction * terrain.material.friction);
113 			collInfo.mat = mat;
114 			collArr.push(collInfo);
115 
116 			info.body0.collisions.push(collInfo);
117 			info.body1.collisions.push(collInfo);
118 		}
119 	};
120 	
121 	jigLib.CollDetectCapsuleTerrain=CollDetectCapsuleTerrain;
122 	
123 })(jigLib);
