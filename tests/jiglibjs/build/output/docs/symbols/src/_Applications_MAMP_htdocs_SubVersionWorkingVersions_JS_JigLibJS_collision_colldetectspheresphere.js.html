 21 (function(jigLib){
 22 	var Vector3DUtil=jigLib.Vector3DUtil;
 23 	var JMatrix3D=jigLib.JMatrix3D;
 24 	var JNumber3D=jigLib.JNumber3D;
 25 	var JConstraint=jigLib.JConstraint;
 26 	var JConfig=jigLib.JConfig;
 27 	var JSphere=jigLib.JSphere;
	var MaterialProperties=jigLib.MaterialProperties;
	var CollPointInfo=jigLib.CollPointInfo;
	var CollisionInfo=jigLib.CollisionInfo;
 28 	 
 29 	/**
 30 	 * @author Muzer(muzerly@gmail.com)
 31 	 * 
 32 	 * @name CollDetectSphereSphere
 33 	 * @class CollDetectSphereSphere handles collisions between spheres 
 34 	 * @extends CollDetectFunctor
 35 	 * @requires CollDetectInfo
 36 	 * @requires CollPointInfo
 37 	 * @requires CollisionInfo
 38 	 * @requires Vector3DUtil
 39 	 * @requires JNumber3D
 40 	 * @requires JMatrix3D
 41 	 * @requires MaterialProperties
 42 	 * @constructor
 43 	 **/
 44 	var CollDetectSphereSphere=function(){
 45 		this.name = "SphereSphere";
 46 		this.type0 = "SPHERE";
 47 		this.type1 = "SPHERE";
 48 	};
 49 	jigLib.extend(CollDetectSphereSphere,jigLib.CollDetectFunctor);
 50 	
 51 	/**
 52 	 * @function collDetect detects a collision and updates the info parameter
 53 	 * @param {CollDetectInfo} info
 54 	 * @param {array} collArray
 55 	 * @type void
 56 	 **/
 57 	CollDetectSphereSphere.prototype.collDetect=function(info, collArr){
 58 		var sphere0 = info.body0;
 59 		var sphere1 = info.body1;

 60 		var oldDelta = Vector3DUtil.subtract(sphere0.get_oldState().position, sphere1.get_oldState().position);
 61 		var newDelta = Vector3DUtil.subtract(sphere0.get_currentState().position, sphere1.get_currentState().position);
 62 
 63 		var oldDistSq = Vector3DUtil.get_lengthSquared(oldDelta);
 64 		var newDistSq = Vector3DUtil.get_lengthSquared(newDelta);
 65 		var radSum = sphere0.get_radius() + sphere1.get_radius();
 66 
 67 		if (Math.min(oldDistSq, newDistSq) < Math.pow(radSum + JConfig.collToll, 2)){
 68 			var oldDist = Math.sqrt(oldDistSq);
 69 			var depth = radSum - oldDist;
 70 			if (oldDist > JNumber3D.NUM_TINY){
 71 				oldDelta = JNumber3D.getDivideVector(oldDelta, oldDist);
 72 			}else{
 73 				oldDelta = Vector3DUtil.Y_AXIS;
 74 				JMatrix3D.multiplyVector(JMatrix3D.getRotationMatrix(0, 0, 1, 360 * Math.random()), oldDelta);
 75 			}
 76 
 77 			var worldPos = Vector3DUtil.add(sphere1.get_oldState().position, JNumber3D.getScaleVector(oldDelta, sphere1.get_radius() - 0.5 * depth));
 78 
 79 			var collPts = [];
 80 			var cpInfo = new CollPointInfo();
 81 			cpInfo.r0 = Vector3DUtil.subtract(worldPos, sphere0.get_oldState().position);
 82 			cpInfo.r1 = Vector3DUtil.subtract(worldPos, sphere1.get_oldState().position);
 83 			cpInfo.initialPenetration = depth;
 84 			collPts.push(cpInfo);
 85 
 86 			var collInfo = new CollisionInfo();
 87 			collInfo.objInfo = info;
 88 			collInfo.dirToBody = oldDelta;
 89 			collInfo.pointInfo = collPts;
 90 
 91 			var mat = new MaterialProperties();
 92 			mat.set_restitution(Math.sqrt(sphere0.get_material().get_restitution() * sphere1.get_material().get_restitution()));
 93 			mat.set_friction(Math.sqrt(sphere0.get_material().get_friction() * sphere1.get_material().get_friction()));
 94 			collInfo.mat = mat;
 95 			collArr.push(collInfo);
 96 
 97 			info.body0.collisions.push(collInfo);
 98 			info.body1.collisions.push(collInfo);
 99 		}
100 	};
101 	
102 	jigLib.CollDetectSphereSphere=CollDetectSphereSphere;
103 	
104 })(jigLib);
