 13 	* Initializes a new CollisionSystem which uses a grid to speed up collision detection.
 14 	* Use this system for larger scenes with many objects.
 15 	* @param sx start point of grid in X axis.
 16 	* @param sy start point of grid in Y axis.
 17 	* @param sz start point of grid in Z axis.
 18 	* @param nx Number of GridEntries in X Direction.
 19 	* @param ny Number of GridEntries in Y Direction.
 20 	* @param nz Number of GridEntries in Z Direction.
 21 	* @param dx Size of a single GridEntry in X Direction.
 22 	* @param dy Size of a single GridEntry in Y Direction.
 23 	* @param dz Size of a single GridEntry in Z Direction.
 24 	*/
 25 	var CollisionSystemGrid=function(sx, sy, sz, nx, ny, nz, dx, dy, dz){
 26 		this.Super();
 27 		if(sx==undefined) sx=0;
 28 		if(sy==undefined) sy=0;
 29 		if(sz==undefined) sz=0;
 30 		if(nx==undefined) nx=20; 
 31 		if(ny==undefined) ny=20;
 32 		if(nz==undefined) nz=20;
 33 		if(dx==undefined) dx=200;
 34 		if(dy==undefined) dy=200;
 35 		if(dz==undefined) dz=200;
 36 		this.nx = nx; this.ny = ny; this.nz = nz;
 37 		this.dx = dx; this.dy = dy; this.dz = dz;
 38 		this.sizeX = nx * dx;
 39 		this.sizeY = ny * dy;
 40 		this.sizeZ = nz * dz;
 41 		this.minDelta = Math.min(dx, dy, dz);
 42                         
 43 		this.startpoint = [sx, sy, sz];
 44 		
 45 		this.gridEntries = [];
 46 		//gridBoxes = new Vector.<JAABox>(nx*ny*nz,true);
 47                         
 48 		var len=this.gridEntries.length;
 49 		for (var j = 0; j < len; ++j){
 50 			var gridEntry = new CollisionSystemGridEntry(null);
 51 			gridEntry.gridIndex = j;
 52 			this.gridEntries[j]=gridEntry;
 53 		}
 54                         
 55 		this.overflowEntries = new CollisionSystemGridEntry(null);
 56 		this.overflowEntries.gridIndex = -1;
 57 	}

 80         CollisionSystemGrid.prototype.calcIndex=function(i, j, k){
 81 		var _i = i % this.nx;
 82 		var _j = j % this.ny;
 83 		var _k = k % this.nz;
 84                         
 85 		return (_i + this.nx * _j + (this.nx + this.ny) * _k);
 86 	};

 89         CollisionSystemGrid.prototype.calcGridForSkin3=function(colBody){
 90 		var i;var j;var k;
 91 		var sides = colBody.get_boundingBox().get_sideLengths();
 92                         
 93 		if ((sides[0] > this.dx) || (sides[1] > this.dy) || (sides[2] > this.dz)){
 94 			//trace("calcGridForSkin3 -- Rigidbody to big for gridsystem - putting it into overflow list (lengths,type,id):", sides.x,sides.y,sides.z,colBody.type,colBody.id,colBody.boundingBox.minPos,colBody.boundingBox.maxPos);
 95 			i = j = k = -1;
 96 			return [i,j,k];
 97 		}
 98 		//trace(sides.x,sides.y,sides.z);
 99                         
100 		var min = colBody.get_boundingBox().get_minPos();
101 		min[0] = JMath3D.wrap(min[0], this.startpoint[0], this.startpoint[0]+this.sizeX);
102 		min[1] = JMath3D.wrap(min[1], this.startpoint[1], this.startpoint[1]+this.sizeY);
103 		min[2] = JMath3D.wrap(min[2], this.startpoint[2], this.startpoint[2]+this.sizeZ);
104                         
105 		i = ( ((min[0]-this.startpoint[0]) / this.dx) % this.nx)|0;
106 		j = ( ((min[1]-this.startpoint[1]) / this.dy) % this.ny)|0;
107 		k = ( ((min[2]-this.startpoint[2]) / this.dz) % this.nz)|0;
108                         
109 		return [i,j,k];
110 	};

113 	CollisionSystemGrid.prototype.calcGridForSkin6=function(colBody){
114 		var tempStoreObject = {};
115 		var i;var j;var k;
116 		var fi;var fj;var fk;
117                         
118 		var sides = colBody.get_boundingBox().get_sideLengths();
119                         
120 		if ((sides[0] > this.dx) || (sides[1] > this.dy) || (sides[2] > this.dz)){
121 			//trace("calcGridForSkin6 -- Rigidbody to big for gridsystem - putting it into overflow list (lengths,type,id):", sides.x,sides.y,sides.z,colBody.type,colBody.id,colBody.boundingBox.minPos,colBody.boundingBox.maxPos);
122 			i = j = k = -1;
123 			fi = fj = fk = 0.0;
124 			tempStoreObject.i = i; tempStoreObject.j = j; tempStoreObject.k = k; tempStoreObject.fi = fi; tempStoreObject.fj = fj; tempStoreObject.fk = fk;
125 			return tempStoreObject;
126 		}
127                         
128 		var min = colBody.get_boundingBox().get_minPos();
129 	
130 		min[0] = JMath3D.getLimiteNumber(min[0], this.startpoint[0], this.startpoint[0] + this.sizeX);
131 		min[1] = JMath3D.getLimiteNumber(min[1], this.startpoint[1], this.startpoint[1] + this.sizeY);
132 		min[2] = JMath3D.getLimiteNumber(min[2], this.startpoint[2], this.startpoint[2] + this.sizeZ);
133                         
134 		fi = (min[0] - this.startpoint[0]) / this.dx;
135 		fj = (min[1] - this.startpoint[1]) / this.dy;
136 		fk = (min[2] - this.startpoint[2]) / this.dz;
137                         
138 		i = fi;
139 		j = fj;
140 		k = fk;
141                         
142 		if (i < 0) { i = 0; fi = 0.0; }
143 		else if (i >= this.nx) { i = 0; fi = 0.0; }
144 		else fi -= i;
145                         
146 		if (j < 0) { j = 0; fj = 0.0; }
147 		else if (j >= this.ny) { j = 0; fj = 0.0; }
148 		else fj -= j;
149                         
150 		if (k < 0) { k = 0; fk = 0.0; }
151 		else if (k >= this.nz) { k = 0; fk = 0.0; }
152 		else fk -= k;
153                         
154 		tempStoreObject.i = i; tempStoreObject.j = j; tempStoreObject.k = k; tempStoreObject.fi = fi; tempStoreObject.fj = fj; tempStoreObject.fk = fk;
155 		//trace(i,j,k,fi,fj,fk);
156 		//trace(colBody.x,colBody.y,colBody.z);
157 		return tempStoreObject;
158 	};

170 		if (!this.findBody(body)) this.collBody.push(body);
171

233         CollisionSystemGrid.prototype.getListsToCheck=function(colBody){
234 		var entries = []; 
235                         
236 		var entry = colBody.externalData;
237 		if (entry == null){
238 			//trace("Warning skin has grid entry null!");
239 			return null;
240 		}
241                         
242 		// todo - work back from the mGridIndex rather than calculating it again...
243 		var i; var j; var k;
244 		var fi; var fj; var fk;
245 		var tempStoreObject = this.calcGridForSkin6(colBody);
246 		i = tempStoreObject.i; j = tempStoreObject.j; k = tempStoreObject.k; fi = tempStoreObject.fi; fj = tempStoreObject.fj; fk = tempStoreObject.fk;
247                         
248 		if (i == -1){
249 			//trace("ADD ALL!");
250 			entries=this.gridEntries.concat();
251 			entries.push(this.overflowEntries);
252 			return entries;
253 		}
254                         
255 		// always add the overflow
256 		entries.push(this.overflowEntries);
257                         
258 		var delta = colBody.get_boundingBox().get_sideLengths(); // skin.WorldBoundingBox.Max - skin.WorldBoundingBox.Min;
259 		var maxI = 1, maxJ = 1, maxK = 1;
260 		if (fi + (delta[0] / this.dx) < 1)
261 			maxI = 0;
262 		if (fj + (delta[1] / this.dy) < 1)
263 			maxJ = 0;
264 		if (fk + (delta[2]/ this.dz) < 1)
265 			maxK = 0;
266                         
267 		// now add the contents of all grid boxes - their contents may extend beyond the bounds
268 		for (var di = -1; di <= maxI; ++di){
269 			for (var dj = -1; dj <= maxJ; ++dj){
270 				for (var dk = -1; dk <= maxK; ++dk){
271 					var thisIndex = this.calcIndex(this.nx + i + di, this.ny + j + dj, this.nz + k + dk); // + ((nx*ny*nz)*0.5);
272 					//trace("ge", gridEntries.length);
273 					if (this.gridEntries.length-1 > thisIndex && thisIndex >=0) {
274 						var start = this.gridEntries[thisIndex];
275                                                 
276 						//trace(thisIndex,gridEntries.length);
277 						if (start != null && start.next != null){
278 							entries.push(start);
279 						}
280 					}
281 				}
282 			}
283 		}
284 		return entries;
285 	}

288         CollisionSystemGrid.prototype.detectAllCollisions=function(bodies, collArr) {
289 		var info;
290 		var fu;
291 		var bodyID;
292 		var bodyType;
293 		this._numCollisionsChecks = 0;
294                         
295 		for(var j=0;j<bodies.length;j++){
296 			var body=bodies[j];
297 			if (!body.isActive) continue;
298 
299 			bodyID = body.id;
300 			bodyType = body.get_type();
301                                 
302 			var lists=this.getListsToCheck(body);
303                                 
304 			for(var k=0;k<lists.length;k++){
305 				var entry=lists[k];                                      
306 				for (entry = entry.next; entry != null; entry = entry.next){
307 					if (body == entry.collisionBody) continue;
308                                                 
309 					if (entry.collisionBody && entry.collisionBody.isActive && bodyID > entry.collisionBody.id)
310 						continue;
311                                                 
312 					if (this.checkCollidables(body, entry.collisionBody) && this.detectionFunctors[bodyType + "_" + entry.collisionBody.get_type()] != undefined){
313 						info = new CollDetectInfo();
314 						info.body0 = body;
315 						info.body1 = entry.collisionBody;
316 						fu = this.detectionFunctors[info.body0.get_type() + "_" + info.body1.get_type()];
317 						fu.collDetect(info, collArr);
318 						this._numCollisionsChecks += 1;
319 					} //check collidables
320 				}// loop over entries
321 			} // loop over lists
322 		} // loop over bodies
323 	}
